 // https://seller.taototo.cn/yp-api/ticket/order/detail
    async function onRequest(context, request) {

        return request
    }

    async function onResponse(context, request, response) {
        if (response.body) {
            var body = JSON.parse(response.body)
            if (body.data.result.unitPrice) {
                var price = body.data.order.initPrice / body.data.order.orderNum
                var unitPrice = body.data.result.unitPrice
                var discount = (price - unitPrice) / 100
                body.data.order.cinemaInfo.movieName = body.data.order.cinemaInfo.movieName + '(' + (body.data.result.unitPrice / 100).toFixed(2) + ' + ' + discount.toFixed(2) +  ')'
            }
            response.body = JSON.stringify(body)
        }
        return response
    }
